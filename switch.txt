class switch1
{
    public static void main (String args[])
    {
        switch (2)
        {
            case 1:
            {
                System.out.println("Monday");
                break;
            }
            case 2 :
            {
                System.out.println("Sunday");
                break;
            }
            default:
            {
                System.out.println("Invalid");
                break;
            }

        }
    }
}
